<script type="text/babel">
  const { Component } = React;
  const { Button, Modal, Input, } = antd;
  class App extends React.Component {
    state = {
      visible: false
    }
    showModal = () => {
      this.setState({visible: true});
    };
    handleOk = e => {
      this.setState({visible: true});
    };
    handleCancel = e => {
      this.setState({visible: true});
    };
    render() {
      return (
        <div>
          <Button type='primary' onClick={this.showModal}>按钮</Button>
          <Modal
            title="标题"
            visible={this.state.visible}
            onOk={this.handleOk}
            onCancel={this.handleCancel}
          >
            <p>内容</p>
          </Modal>
        </div >
      )
    }
  }

  ReactDOM.render(
    <App />,
    document.getElementById("app")
  )

</script>
